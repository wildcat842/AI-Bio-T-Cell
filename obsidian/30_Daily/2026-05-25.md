### 1.2 Seurat 시각화 (R 스크립트)
```R
# notebooks/R/01_first_visualization.R 만들기
suppressPackageStartupMessages({
  library(Seurat)
  library(ggplot2)
  library(patchwork)
})

setwd("~/work/AI-Bio-T-Cell")
out_dir <- "reports/figures/2026-05-25_first_exploration"
dir.create(out_dir, recursive = TRUE, showWarnings = FALSE)

# 가장 작은 .rds 로드 (검증용)
rds_files <- list.files("datasets/01_reaptec_atlas/processed/seurat_objects/",
                         pattern="\\.[rR]ds$", recursive=TRUE, full.names=TRUE)
sizes <- file.info(rds_files)$size
target <- rds_files[which.min(sizes)]
cat("Loading:", target, sprintf("(%.2f MB)\n", file.info(target)$size/1e6))

obj <- readRDS(target)
print(obj)
cat("Metadata cols:", paste(colnames(obj@meta.data), collapse=", "), "\n")

# === Plot 1: UMAP 통합 비교 ===
p_raw     <- DimPlot(obj, reduction="umap.unintegrated", group.by="orig.ident") + 
             ggtitle("Unintegrated") + theme(legend.position="bottom")
p_harmony <- DimPlot(obj, reduction="umap.harmony",      group.by="orig.ident") + 
             ggtitle("Harmony")      + theme(legend.position="bottom")
p_cca     <- DimPlot(obj, reduction="umap.cca",          group.by="orig.ident") + 
             ggtitle("CCA")          + theme(legend.position="bottom")
combo <- p_raw | p_harmony | p_cca

ggsave(file.path(out_dir, "01_integration_comparison.png"),
       combo, width=15, height=5, dpi=300, bg="white")
ggsave(file.path(out_dir, "01_integration_comparison.pdf"),
       combo, width=15, height=5)
cat("[OK] saved: 01_integration_comparison.png/pdf\n")

# === Plot 2: T cell markers ===
markers <- c("CD3D", "CD8A", "CD4", "FOXP3", "PDCD1", "TCF7")
markers_present <- intersect(markers, rownames(obj))
if (length(markers_present) > 0) {
  p <- FeaturePlot(obj, features=markers_present, reduction="umap.harmony",
                   ncol=3, order=TRUE) +
       plot_annotation(title="T cell marker expression (Harmony UMAP)")
  ggsave(file.path(out_dir, "02_marker_featureplot.png"),
         p, width=12, height=8, dpi=300, bg="white")
  cat("[OK] saved: 02_marker_featureplot.png (markers:", 
      paste(markers_present, collapse=", "), ")\n")
}

# === Plot 3: cluster/celltype 분포 ===
if ("seurat_clusters" %in% colnames(obj@meta.data)) {
  p <- DimPlot(obj, reduction="umap.harmony", group.by="seurat_clusters",
               label=TRUE, repel=TRUE) + ggtitle("Seurat clusters")
  ggsave(file.path(out_dir, "03_clusters.png"), p, width=8, height=6, dpi=300, bg="white")
}

cat("\n=== Done ===\n")
cat("Outputs at:", out_dir, "\n")
list.files(out_dir)
```

### 빠른 비교 (R obj shape vs Python AnnData shape)
```bash
# R 측 shape
Rscript -e '
obj <- readRDS(list.files("datasets/01_reaptec_atlas/processed/seurat_objects",
       pattern="\\.[rR]ds$", recursive=TRUE, full.names=TRUE)[1])
cat("R:    cells x genes =", ncol(obj), "x", nrow(obj), "\n")
'

# Python 측 shape
python -c "
import anndata as ad, pathlib
f = sorted(pathlib.Path('datasets/01_reaptec_atlas/processed/h5ad').glob('*.h5ad'))[0]
a = ad.read_h5ad(f, backed='r')
print(f'Py:   cells x genes = {a.shape[0]} x {a.shape[1]}')
print(f'      obs cols      = {len(a.obs.columns)} ({list(a.obs.columns)[:5]}...)')
print(f'      X type        = {type(a.X).__name__}')
"
```

→ R과 Python의 cells × genes가 일치하면 변환 성공.
